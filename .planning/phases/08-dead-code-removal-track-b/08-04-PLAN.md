```cmake
find_package(Qt3 REQUIRED)

file(GLOB SHARED_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/src/shared/*.cpp")
file(GLOB WIN32_SOURCES  "${CMAKE_CURRENT_SOURCE_DIR}/src/win32/*.cpp")
# Locate .ui files — verify location by reading source directory
file(GLOB UI_FILES "${CMAKE_CURRENT_SOURCE_DIR}/src/shared/*.ui")

# List Q_OBJECT headers explicitly — read src/shared/*.h and src/win32/*.h to identify
set(MOC_HEADERS
    # populate after reading source directory
)

qt3_wrap_ui(GENERATED_UI ${UI_FILES})
qt3_wrap_cpp(GENERATED_MOC ${MOC_HEADERS})

add_executable(LocalizationTool
    ${SHARED_SOURCES}
    ${WIN32_SOURCES}
    ${GENERATED_UI}
    ${GENERATED_MOC}
)

target_include_directories(LocalizationTool PRIVATE
    ${CMAKE_CURRENT_BINARY_DIR}           # ui_*.h generated files
    ${QT3_INCLUDE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/src/shared
    ${CMAKE_CURRENT_SOURCE_DIR}/src/win32
    ${SWG_OURS}/localization/include
    ${SWG_OURS}/localizationArchive/include/public
    ${SWG_OURS}/unicode/include
    ${SWG_OURS}/unicodeArchive/include/public
    ${SWG_OURS}/archive/include
    ${SWG_ENGINE_SOURCE_DIR}/shared/library/sharedFoundation/include/public
    ${SWG_ENGINE_SOURCE_DIR}/shared/library/sharedDebug/include/public
    ${SWG_ENGINE_SOURCE_DIR}/shared/library/sharedFile/include/public
    ${SWG_ENGINE_SOURCE_DIR}/shared/library/sharedThread/include/public
    ${SWG_ENGINE_SOURCE_DIR}/shared/library/sharedCompression/include/public
)

target_link_libraries(LocalizationTool PRIVATE
    ${QT3_LIBRARIES}
    localization
    localizationArchive
    unicode
    unicodeArchive
    archive
    sharedFoundation
    sharedDebug
    sharedFile
    sharedThread
    sharedCompression
    ${STLPORT_LIBRARY}
    legacy_stdio_definitions
)

target_precompile_headers(LocalizationTool PRIVATE src/shared/FirstLocalizationTool.h)

set(_QT3_DLL "${SWG_EXTERNALS_FIND}/qt/3.3.4/lib/qt-mt334.dll")
add_custom_command(TARGET LocalizationTool POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${_QT3_DLL}" "$<TARGET_FILE_DIR:LocalizationTool>/qt-mt334.dll"
    VERBATIM
)
```
