```cpp
        D3D11_SDK_VERSION, &device, &fl, &context);

FATAL(fl < D3D_FEATURE_LEVEL_11_0,
```

```cpp
void Direct3d11_Device::displayModeChanged()
{
    // Recreate swap chain at new HWND client size
    if (!ms_installed) return;
    
    RECT rc;
    GetClientRect(ms_window, &rc);
    int newWidth  = rc.right - rc.left;
    int newHeight = rc.bottom - rc.top;
    if (newWidth == ms_width && newHeight == ms_height) return;
    
    DEBUG_REPORT_LOG_PRINT(true, ("Direct3d11_Device: displayModeChanged %dx%d -> %dx%d\n", ms_width, ms_height, newWidth, newHeight));
    
    // Release back-buffer references
    ms_depthStencilDSV.Reset();
    ms_depthStencilTex.Reset();
    ms_backBufferRTV.Reset();
    ms_context->ClearState();
    
    // ResizeBuffers
    HRESULT hr = ms_swapChain->ResizeBuffers(0, newWidth, newHeight, DXGI_FORMAT_UNKNOWN, 0);
    FATAL_DX_HR("IDXGISwapChain1::ResizeBuffers failed", hr);
    
    ms_width  = newWidth;
    ms_height = newHeight;
    
    // Recreate back-buffer RTV + depth-stencil at new size (same code as create())
    ComPtr<ID3D11Texture2D> backBuffer;
    ms_swapChain->GetBuffer(0, IID_PPV_ARGS(&backBuffer));
    ms_device->CreateRenderTargetView(backBuffer.Get(), nullptr, &ms_backBufferRTV);
    
    D3D11_TEXTURE2D_DESC dsDesc = {};
    dsDesc.Width = newWidth; dsDesc.Height = newHeight; dsDesc.MipLevels = 1; dsDesc.ArraySize = 1;
    dsDesc.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
    dsDesc.SampleDesc.Count = 1;
    dsDesc.Usage = D3D11_USAGE_DEFAULT;
    dsDesc.BindFlags = D3D11_BIND_DEPTH_STENCIL;
    ms_device->CreateTexture2D(&dsDesc, nullptr, &ms_depthStencilTex);
    ms_device->CreateDepthStencilView(ms_depthStencilTex.Get(), nullptr, &ms_depthStencilDSV);
}
```
