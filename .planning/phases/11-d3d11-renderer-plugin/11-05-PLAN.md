**D3DCompile usage pattern** (RESEARCH §Architecture Patterns Pattern 3, lines 270-328 — full code reproduced for executor convenience):

```cpp
#include <d3dcompiler.h>

struct CachedShader {
    std::vector<uint8_t> bytecode;
    uint64_t             sourceHash;
};

CachedShader compileOrLoadVS(char const * sourceText, size_t sourceLen,
                             D3D_SHADER_MACRO const * defines)
{
    uint64_t hash = computeHash(sourceText, sourceLen, defines);
    std::string cachePath = std::format("{}/{:016x}.cso", ms_shaderCacheDir, hash);

    CachedShader out;
    if (loadFromDisk(cachePath, out) && out.sourceHash == hash)
        return out;   // cache hit

    ComPtr<ID3DBlob> bytecode, errors;
    UINT flags = D3DCOMPILE_ENABLE_STRICTNESS;
#ifdef _DEBUG
    flags |= D3DCOMPILE_DEBUG | D3DCOMPILE_SKIP_OPTIMIZATION;
#else
    flags |= D3DCOMPILE_OPTIMIZATION_LEVEL3;
#endif

    HRESULT hr = D3DCompile(
        sourceText, sourceLen,
        "shader",         // virtual filename (for error messages)
        defines,          // mirrors D3DXMACRO array used by D3D9
        D3D_COMPILE_STANDARD_FILE_INCLUDE,   // simple #include resolver
        "main",
        "vs_5_0",         // or "ps_5_0" for pixel shader
        flags, 0,
        &bytecode, &errors);
    FATAL(FAILED(hr), ("D3DCompile failed: %s",
        errors ? (char*)errors->GetBufferPointer() : "no error blob"));

    out.bytecode.assign((uint8_t*)bytecode->GetBufferPointer(),
                        (uint8_t*)bytecode->GetBufferPointer() + bytecode->GetBufferSize());
    out.sourceHash = hash;
    writeToDisk(cachePath, out);
    return out;
}

// Then create the shader object:
ComPtr<ID3D11VertexShader> vs;
device->CreateVertexShader(out.bytecode.data(), out.bytecode.size(), nullptr, &vs);
```

**Pitfall 1 — SV_POSITION semantic injection** (RESEARCH lines 408-421):
```cpp
D3D_SHADER_MACRO defines[] = {
    {"POSITION", "SV_POSITION"},   // legacy alias → system-value semantic
    {nullptr, nullptr},
};
```
Pass this defines array into `D3DCompile`. Caveat: if `.vsh` source uses `POSITION` as a vertex INPUT semantic (`struct VertexInput { float4 pos : POSITION; }`), the macro will incorrectly rewrite the input semantic too. Per RESEARCH: verify with a grep over the asset shader catalog during the Wave 4 / Wave 5 boundary. If the asset catalog has both input-POSITION and output-POSITION, use a more targeted approach: don't use the macro, instead pre-process shader source text to substitute only `: POSITION` in output struct field positions (regex `: POSITION(\s*;|\s*,\s*float)` → `: SV_POSITION$1`).

**Pitfall 2 — cbuffer 16-byte alignment** (RESEARCH lines 423-429):
Always use `DirectX::XMFLOAT4X4` and `DirectX::XMFLOAT4` (or pad with `float4 _padN`) on C++ side. Mirror HLSL `cbuffer` declaration exactly. `static_assert(sizeof(MyCB) == sizeof_in_hlsl)` on every struct.

Example pattern (from PATTERNS §`Direct3d11_ConstantBuffer.{h,cpp}` lines 534-546 + RESEARCH Pattern 2 lines 236-269):
```cpp
// HLSL (.vsh source):
cbuffer PerObject : register(b0)
{
    float4x4 worldViewProj;
    float4   userConstants[8];
};

// C++ mirror:
struct PerObjectCB {
    DirectX::XMFLOAT4X4 worldViewProj;     // 64 bytes
    DirectX::XMFLOAT4   userConstants[8];  // 128 bytes
};
static_assert(sizeof(PerObjectCB) == 192, "PerObjectCB size mismatch with HLSL cbuffer");
static_assert(sizeof(PerObjectCB) % 16 == 0, "PerObjectCB not 16-byte aligned");
```

`install(cacheDir)`: ensure directory exists (`std::filesystem::create_directories`); store path. Log `Direct3d11_ShaderCache: install at <path>`.

`hashSource(text, length, defines)`: use FNV-1a 64-bit (non-cryptographic, fast, deterministic, no dependency). Hash text bytes first; then for each `D3D_SHADER_MACRO` in defines (until `{nullptr,nullptr}` terminator), hash `Name` then `\0` then `Definition` then `\0` (so `{POSITION,SV_POSITION}` and `{POSITION,POSITION}` produce distinct hashes).

`tryLoad(hash, outBlob)`:
1. Build path: `ms_cacheDir + "/" + hex(hash) + ".cso"` (use 16-hex-digit lowercase)
2. `fopen(path, "rb")` — if fails, return false (miss); `ms_misses++`
3. `fseek/ftell` for size; allocate via `D3DCreateBlob(size, &outBlob)`; `fread(outBlob->GetBufferPointer(), size, 1, f)`; `fclose`
4. Return true; `ms_hits++`

`store(hash, blob)`:
1. Build same path
2. `fopen(path, "wb")` — if fails (e.g., permission error), log a warning but do NOT FATAL (cache is opportunistic, not contract); next launch just recompiles
3. `fwrite(blob->GetBufferPointer(), blob->GetBufferSize(), 1, f)`; `fclose`

**Security note (per RESEARCH §Security Domain V12):** validate hash is hex-only before path concat (it's our own output but defense-in-depth is cheap). Cache dir is engine-configured via ConfigDirect3d11; never user-input.

`remove()`: log hit/miss counts. No file cleanup (cache survives across launches per D-03).

**Implementation** — the generator reads the engine-side `Stage` struct (the same one `Direct3d9_ShaderImplementationData::Stage::construct` reads at `Direct3d9_ShaderImplementationData.cpp:138-164`) and synthesizes `ps_5_0` HLSL source.

```cpp
namespace Direct3d11_FfpGeneratorNamespace
{
    struct TupleKey {
        // Captures the SET of (colorOp, alphaOp, colorArgs, alphaArgs) across all stages
        // SHA1 or FNV-128 of the packed stage descriptors
        uint64_t hash;
        bool operator<(TupleKey const &rhs) const { return hash < rhs.hash; }
    };
    
    std::map<TupleKey, Microsoft::WRL::ComPtr<ID3D11PixelShader>> ms_cache;
}

static std::string generateHlsl(ShaderImplementation::Pass::Stage const *stages, int n)
{
    // Cover at minimum: MODULATE, ADD, SELECTARG1 per SPEC R4 acceptance
    std::string src;
    src += "// Auto-generated FFP equivalent\n";
    src += "Texture2D tex0 : register(t0);\n";
    src += "SamplerState samp0 : register(s0);\n";
    src += "struct PSInput {\n";
    src += "    float4 pos : SV_POSITION;\n";
    src += "    float4 color : COLOR;\n";
    src += "    float2 uv : TEXCOORD0;\n";
    src += "};\n";
    src += "float4 main(PSInput input) : SV_TARGET {\n";
    src += "    float4 t = tex0.Sample(samp0, input.uv);\n";
    
    for (int i = 0; i < n; ++i) {
        auto const &s = stages[i];
        // Read s.m_colorOperation (a TextureOperation enum — values match D3DTOP_MODULATE/ADD/SELECTARG1/DISABLE)
        switch (static_cast<int>(s.m_colorOperation)) {
            case /* MODULATE */ 4: src += "    t = t * input.color;\n"; break;
            case /* ADD */      7: src += "    t = t + input.color;\n"; break;
            case /* SELECTARG1 */ 2: src += "    t = t;\n"; break;
            case /* DISABLE */  1: /* skip */ break;
            default:
                FATAL(true, ("Direct3d11_FfpGenerator: unsupported COLOROP %d at stage %d (SPEC R4 covers MODULATE/ADD/SELECTARG1 minimum; extend generator if asset uses others)",
                    static_cast<int>(s.m_colorOperation), i));
        }
        // Same dispatch for s.m_alphaOperation if non-DISABLE
    }
    
    src += "    return t;\n";
    src += "}\n";
    return src;
}

ID3D11PixelShader * Direct3d11_FfpGenerator::generateOrLookup(
    ShaderImplementation::Pass::Stage const *stages, int n)
{
    TupleKey key = {};
    // Hash the stage tuple — color/alpha ops + args across all n stages
    key.hash = Direct3d11_ShaderCache::hashSource(
        reinterpret_cast<char const*>(stages),
        n * sizeof(ShaderImplementation::Pass::Stage),
        nullptr);
    
    auto it = ms_cache.find(key);
    if (it != ms_cache.end()) return it->second.Get();
    
    // Generate HLSL + compile via D3DCompile + cache resulting ID3D11PixelShader
    std::string hlsl = generateHlsl(stages, n);
    
    ComPtr<ID3DBlob> blob, errors;
    UINT flags = D3DCOMPILE_ENABLE_STRICTNESS;
    #ifdef _DEBUG
        flags |= D3DCOMPILE_DEBUG | D3DCOMPILE_SKIP_OPTIMIZATION;
    #else
        flags |= D3DCOMPILE_OPTIMIZATION_LEVEL3;
    #endif
    
    HRESULT hr = D3DCompile(hlsl.c_str(), hlsl.size(),
        "ffp_generated", nullptr, nullptr,
        "main", "ps_5_0", flags, 0, &blob, &errors);
    FATAL(FAILED(hr), ("Direct3d11_FfpGenerator: D3DCompile failed: %s",
        errors ? (char*)errors->GetBufferPointer() : "no error blob"));
    
    ComPtr<ID3D11PixelShader> shader;
    hr = Direct3d11_Device::getDevice()->CreatePixelShader(
        blob->GetBufferPointer(), blob->GetBufferSize(), nullptr, &shader);
    FATAL_DX_HR("Direct3d11_FfpGenerator: CreatePixelShader failed", hr);
    
    ms_cache[key] = shader;
    return shader.Get();
}
```
