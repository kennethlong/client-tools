2. Author `std_hash_dump.cpp` — a 30-line helper that reads `known-strings.txt` line by line and writes `<input>\t0x<crc-hex>\t0x<std-hash-hex>` per line:

```cpp
// .planning/phases/09-stlport-msvc-stl/baseline/std_hash_dump.cpp
// Phase 9 Wave 0 — emit Crc::calculateLowerString + std::hash<std::string> for a fixed
// input list. Output format: "<input>\t0x<crc-hex>\t0x<std-hash-hex>" per line.
//
// Build target: validation_helper_std_hash_dump (gated behind WHITENGOLD_BUILD_VALIDATION_HELPERS).
// Run: validation_helper_std_hash_dump <known-strings.txt> > before-crc.txt
//
// This file is intentionally a flat C++ console exe with no PCH and minimal dependencies.

#include "sharedFoundation/Crc.h"
#include <cstdint>
#include <cstdio>
#include <fstream>
#include <functional>
#include <iomanip>
#include <iostream>
#include <sstream>
#include <string>

int main(int argc, char *argv[]) {
    if (argc < 2) { std::fprintf(stderr, "usage: %s <input-list>\n", argv[0]); return 1; }
    std::ifstream in(argv[1]);
    if (!in) { std::fprintf(stderr, "cannot open %s\n", argv[1]); return 2; }

    std::hash<std::string> stdHasher;
    std::string line;
    while (std::getline(in, line)) {
        if (line.empty()) continue;
        const std::uint32_t crc = Crc::calculateLowerString(line.c_str());
        const std::size_t  sh  = stdHasher(line);
        std::cout << line
                  << "\t0x" << std::hex << std::setw(8) << std::setfill('0') << crc
                  << "\t0x" << std::hex << sh
                  << std::dec << "\n";
    }
    return 0;
}
```

   IMPORTANT: If `Crc::calculateLowerString` is not the exact symbol name, read `Crc.h` to find the right one (alternatives in this codebase: `Crc::calculate`, `Crc::calculateWithToLower`, or `CrcLowerString::calculateCrc`). Use the SAME function whose output is persisted in TRE asset hashes — the goal is to capture the persistent CRC. RESEARCH.md uses `Crc::calculateWithToLower` in prose; verify the actual symbol against `Crc.h`.

// .planning/phases/09-stlport-msvc-stl/baseline/datatable_round_trip.cpp
// Phase 9 Wave 0 — D-06 round-trip persist baseline.
// Loads a retail DataTable IFF and dumps every cell as text.
// Format: "<row>\t<col-index>\t<col-name>\t<type>\t<value>" per line.
//
// Build target: validation_helper_datatable_round_trip (gated behind
// WHITENGOLD_BUILD_VALIDATION_HELPERS).
// Run: validation_helper_datatable_round_trip <path-to-iff> > before-datatable-round-trip.txt
//
// Recommendation: misc/quest_task_data.iff (small, ~20 columns; in retail TRE).

#include "sharedFile/Iff.h"
#include "sharedFoundation/SetupSharedFoundation.h"
#include "sharedFoundation/ExitChain.h"
#include "sharedUtility/DataTable.h"
#include <cstdio>
#include <iostream>
#include <string>

int main(int argc, char *argv[]) {
    if (argc < 2) { std::fprintf(stderr, "usage: %s <datatable.iff>\n", argv[0]); return 1; }

    SetupSharedFoundation::Data setupData(SetupSharedFoundation::Data::D_console);
    SetupSharedFoundation::install(setupData);

    Iff iff(argv[1]);
    DataTable dt;
    dt.load(iff);

    const int nrows = dt.getNumRows();
    const int ncols = dt.getNumColumns();
    for (int row = 0; row < nrows; ++row) {
        for (int col = 0; col < ncols; ++col) {
            const std::string &cname = dt.getColumnName(col);
            const int type = static_cast<int>(dt.getDataTypeForColumn(col));
            std::cout << row << "\t" << col << "\t" << cname << "\t" << type << "\t";
            // Coarse-but-deterministic dump: try string, then int, then float.
            // Real DataTable accessors throw/assert on type mismatch; use type tag to switch.
            // Helper is best-effort — exact type enum values come from DataTableColumnType.
            const char *s = dt.getStringValue(cname.c_str(), row);
            if (s) std::cout << s;
            std::cout << "\n";
        }
    }

    ExitChain::quit();
    return 0;
}
```

   IMPORTANT: If `DataTableColumnType` is in a different header or `dt.getStringValue` doesn't accept any column type, read `DataTable.h` and adjust the cell-dump strategy to call the right typed accessor based on `getDataTypeForColumn`. The exact API may differ; the goal is "deterministic textual dump of every cell." Adjust the helper until it compiles AND produces deterministic output.

// .planning/phases/09-stlport-msvc-stl/baseline/sort_stability_dump.cpp
// Phase 9 Wave 0 — sort stability baseline.
// Pin a known input array, run std::sort with CrcLowerString::LessAbcOrderReferenceComparator,
// dump indexes in sorted order. Acceptance: pre/post Phase 9 dumps match.

#include "sharedFoundation/CrcLowerString.h"
#include <algorithm>
#include <iostream>
#include <vector>

int main() {
    // Fixed input — same set used for before-crc.txt's first 12 entries (any deterministic
    // set works; the goal is comparator stability, not specific strings).
    const char *inputs[] = {
        "appearance/wpn_kashyyykian_axe.apt",
        "creature/badge.iff",
        "terrain/tatooine.trn",
        "misc/quest_task_data.iff",
        "string/en/quest/ground/lok_quest_text.stf",
        "appearance/lair_kashyyykian.apt",
        "appearance/wpn_blaster.apt",
        "creature/dewback.iff",
        "creature/bantha.iff",
        "object/creature/player/human_male.iff",
        "object/tangible/component/weapon/component_weapon_blaster_pistol.iff",
        "object/tangible/loot/loot_collection_lava_skull.iff",
    };
    constexpr int N = sizeof(inputs) / sizeof(inputs[0]);

    std::vector<CrcLowerString> arr;
    arr.reserve(N);
    for (int i = 0; i < N; ++i) arr.emplace_back(inputs[i]);

    std::vector<const CrcLowerString *> ptrs;
    ptrs.reserve(N);
    for (auto &c : arr) ptrs.push_back(&c);

    std::sort(ptrs.begin(), ptrs.end(), CrcLowerString::LessAbcOrderReferenceComparator{});

    for (const auto *p : ptrs) std::cout << p->getString() << "\n";
    return 0;
}
```

   IMPORTANT: If `LessAbcOrderReferenceComparator` is not directly default-constructible, read `CrcLowerString.h` for the actual comparator API. Adjust the sort line accordingly.
