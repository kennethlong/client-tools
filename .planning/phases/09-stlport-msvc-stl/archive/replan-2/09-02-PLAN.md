```cpp
// src/engine/shared/library/sharedFoundation/include/public/sharedFoundation/StlCompat.h
//
// Phase 9 Wave 2 — alias bridge from STLPort hash_map/hash_set names to MSVC's
// std::unordered_map/std::unordered_set. Allows the existing call-sites
// (std::hash_map<K,V>, std::hash_set<K>, std::hash_multimap<K,V>) to compile
// against MSVC STL with zero call-site changes. The cleanup commit at end of
// Wave 6 migrates call-sites from std::hash_* to std::unordered_* directly
// and DELETES this header.
//
// HARD CONSTRAINT (D-02-clarification, from spike 001b INVALIDATED finding):
// DO NOT add #ifdef _STLP_VERSION coexistence guards here. STLPort's
// stl/_epilog.h does `#define std STLPORT` which is a TEXT-LEVEL identifier
// rewrite, not a namespace move. Any branch attempting to bridge STLPort and
// MSVC parsers in the same TU produces silent miscompiles.

#pragma once
#include <unordered_map>
#include <unordered_set>

namespace std {

    template <class K, class V,
              class H = std::hash<K>,
              class E = std::equal_to<K>,
              class A = std::allocator<std::pair<const K, V>>>
    using hash_map = std::unordered_map<K, V, H, E, A>;

    template <class K, class V,
              class H = std::hash<K>,
              class E = std::equal_to<K>,
              class A = std::allocator<std::pair<const K, V>>>
    using hash_multimap = std::unordered_multimap<K, V, H, E, A>;

    template <class K,
              class H = std::hash<K>,
              class E = std::equal_to<K>,
              class A = std::allocator<K>>
    using hash_set = std::unordered_set<K, H, E, A>;

}
```
